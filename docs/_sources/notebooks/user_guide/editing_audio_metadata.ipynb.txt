   ]
  },
  {
   "cell_type": "code",
   "execution_count": 4,

   "outputs": [],
   "source": [
    "def print_metadata(audio_file):\n",
    "    for field, value in audio_file.__dict__.items():\n",
    "        if not field.startswith(\"_\"):\n",
    "            if field in {\"artwork\", \"lyrics\"}:\n",
    "                if value:\n",
    "                    value = type(value)\n",
    "            field = (field.upper() if field == \"isrc\" \n",
    "                     else field.replace(\"_\", \" \").capitalize())\n",
    "            print(f\"{field}: {value}\")"
